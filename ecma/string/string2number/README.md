### [Parser (String2Number)](code.zip)

```js
let char = 'A';
let hexa = char.charCodeAt().toString(16);

console.log(hexa); //=> 41

char = '0x41';
hexa = String.fromCharCode(parseInt(char, 16));

console.log(hexa); //=> A
```
